⚙️ 4. Base App Entry (Vite)
src/js/main.js
import '../css/style.css';
import { initApp } from './app.js';

initApp();
src/js/app.js

This will act like your controller (SleepOutside style).

import { loadHome } from './pages/home.js';
import { loadWeather } from './pages/weather.js';
import { loadPlaces } from './pages/places.js';
import { loadMap } from './pages/map.js';
import { loadFavorites } from './pages/favorites.js';

export function initApp() {
  const page = document.body.dataset.page;

  switch (page) {
    case 'home':
      loadHome();
      break;
    case 'weather':
      loadWeather();
      break;
    case 'places':
      loadPlaces();
      break;
    case 'map':
      loadMap();
      break;
    case 'favorites':
      loadFavorites();
      break;
    default:
      loadHome();
  }
}

🔌 Step 2: API Layer (VERY IMPORTANT)

We simulate a real API like in production.

src/js/api/placesApi.js
import { places } from '../data/mockPlaces.js';

export async function getPlaces() {
  return places;
}

export async function getFeaturedPlaces() {
  return places.filter((place) => place.featured);
}

export async function getPlaceById(id) {
  return places.find((place) => place.id === Number(id));
}

✔ This abstraction is CRITICAL
✔ Later you can plug a real backend without changing UI

🧩 Step 3: Component (Reusable Card)
src/js/components/placeCard.js
export function placeCard(place) {
  return `
    <div class="card">
      <img src="${place.image}" alt="${place.name}" />
      <h3>${place.name}</h3>
      <p>${place.description}</p>
      <button data-id="${place.id}" class="view-btn">
        View Details
      </button>
    </div>
  `;
}

🏠 Step 5: Use Data in Home Page
src/js/pages/home.js
import { getFeaturedPlaces } from '../api/placesApi.js';
import { placeCard } from '../components/placeCard.js';

export async function loadHome() {
  const app = document.querySelector('#app');

  const places = await getFeaturedPlaces();

  app.innerHTML = `
    <h1>Congo Empire 🇨🇩</h1>
    <div id="places-container"></div>
  `;

  const container = document.querySelector('#places-container');

  container.innerHTML = places.map(placeCard).join('');
}
